1）冒泡排序(Bubble Sort)
	
	比较表中的相连元素
		
	public void bubbleSort(int[] arr) {
	    int len = arr.length();
	    for (int i = 0; i < len - 1; i++) {
	        for (var j = 0; j < len - 1 - i; j++) {
	            if (arr[j] > arr[j+1]) {        // 相邻元素两两对比
	                var temp = arr[j+1];        // 元素交换
	                arr[j+1] = arr[j];
	                arr[j] = temp;
	            }
	        }
	    }
	}
